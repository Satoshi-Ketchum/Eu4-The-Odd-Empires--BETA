		NOT = { has_global_flag = BRAKNAMKOLONIIGLOBALFLAGATLANTROPA }

				value = 5
				region = atlantis_region
				is_city = yes
			}
		
        OR = {
            TAG = SIE
            TAG = X33 
        }
    }
	option = {
		name = "atlantropa_sienaalevent.19.a"
        every_owned_province = {
            limit = {
                region = atlantis_region
             }
             add_core = C9F
             cede_province = C9F
        }
        			create_subject = {
				subject_type = commercial_enterprise
				who = C9F
			}
        add_adm_power = 100
        C9F = {
            	change_government = republic
				add_government_reform = trade_company_government
				change_religion = ROOT
				change_primary_culture = ROOT
               define_ruler = {
					change_adm = 3
					change_dip = 3
					change_mil = 3
					max_age = 50
					min_age = 30
				}
                add_adm_power = 100
				add_dip_power = 100
				add_mil_power = 100
				add_treasury = 200
				set_prestige_to_zero = yes
				add_republican_tradition = 100
                
        }
        add_stability = 1
	}
    	option = {
		name = "atlantropa_sienaalevent.19.b"
